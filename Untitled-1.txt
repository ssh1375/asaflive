model Domain {
  id        String     @id @default(dbgenerated("uuid_generate_v7()"))
  name      String     @unique  // e.g. "asaflive.ir"
  userRoles UserRole[]

  createdAt DateTime @default(now())
  updatedAt DateTime @updatedAt
}


model Role {
  id          String       @id @default(dbgenerated("uuid_generate_v7()"))
  name        String       @unique
  permissions Permission[] // implicit M:N
  users   User[]
  domain      Domain   @relation(fields: [domainId], references: [id])
  
  createdAt DateTime @default(now())
  updatedAt DateTime @updatedAt
}


model Permission {
  id    String @id @default(dbgenerated("uuid_generate_v7()"))
  name  String @unique  // e.g. "article:create"
  roles Role[] // implicit M:N


  @@check(name ~ '^[^:]+:[^:]+$')

  createdAt DateTime @default(now())
  updatedAt DateTime @updatedAt
}


model User {
  id              String    @id @default(dbgenerated("uuid_generate_v7()")) @db.Uuid
  email           String?   @unique
  phone           String    @unique
  passwordHash    String
  firstName       String
  lastName        String
  avatarUrl       String?
  isActive        Boolean   @default(true)
  isPhoneVerified Boolean   @default(false)
  lastLoginAt     DateTime?
  roles           Role[]
  createdAt       DateTime  @default(now())
  updatedAt       DateTime  @updatedAt
}
